k_list = [1, 3, 5, 7, 9, 11, 13, 15, 50]

def get_train_test_accuracy(k_list: list=k_list, metric=None):
    train = []
    test = []
    n_k = []

    for k in k_list:
        if metric:
            model = KNeighborsClassifier(n_neighbors=k, metric=metric)
        else:
            model = KNeighborsClassifier(n_neighbors=k)
        model.fit(X, y)
        train.append(model.score(X_train, y_train))
        test.append(model.score(X_test, y_test))
        n_k.append(100/k)
    
    result = pd.DataFrame(
        data = {
            "N/k": n_k, 
            "train error": train, 
            "test error":test
        },
    )
    return result
